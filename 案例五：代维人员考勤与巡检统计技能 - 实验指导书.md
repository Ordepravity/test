# 实验指导书：代维人员考勤与巡检统计技能

## 实验概述

本实验旨在指导学员使用 OpenClaw 平台开发一个代维人员考勤与巡检统计技能，通过该技能自动汇总代维人员的签到记录、巡检完成情况和异常事件上报数据，生成日报/周报报表，帮助代维管理人员快速掌握代维团队的工作状态，提高管理效率。

## 实验目标

1. 掌握 OpenClaw 技能多数据源整合方法
2. 学会构建考勤统计模型
3. 实现巡检任务完成率计算
4. 完成多维度报表生成
5. 支持日期范围筛选和导出

## 实验环境

| 环境项 | 要求 |
|--------|------|
| OpenClaw 平台 | v2.3 以上版本 |
| Python | 3.8 及以上 |
| 数据库 | MySQL 5.7 或 SQLite |
| Excel 支持 | openpyxl |
| 模拟数据 | 包含 10 名代维人员、1周测试数据 |

from dataclasses import dataclass
from datetime import datetime
from typing import List, Optional

@dataclass
class InspectionTask:
    task_id: str
    staff_id: str
    task_type: str           # 基站巡检/故障处理/专项整治
    base_station_id: str
    base_station_name: str
    planned_date: str
    actual_date: Optional[str] = None
    status: str = "pending"   # pending/in_progress/completed/cancelled
    items: List[str] = None
    completed_items: List[str] = None
    remarks: str = ""

    def __post_init__(self):
        if self.items is None:
            self.items = []
        if self.completed_items is None:
            self.completed_items = []

    def completion_rate(self) -> float:
        if not self.items:
            return 100.0 if self.status == "completed" else 0.0
        return round(len(self.completed_items) / len(self.items) * 100, 1)

    def to_dict(self):
        return {
            "task_id": self.task_id,
            "staff_id": self.staff_id,
            "task_type": self.task_type,
            "base_station_id": self.base_station_id,
            "base_station_name": self.base_station_name,
            "planned_date": self.planned_date,
            "actual_date": self.actual_date,
            "status": self.status,
            "items": self.items,
            "completed_items": self.completed_items,
            "completion_rate": self.completion_rate(),
            "remarks": self.remarks

```python
from datetime import datetime, timedelta
from typing import Dict, List
from collections import defaultdict
from models.staff import Staff
from models.check_record import CheckRecord, AttendanceSummary

class AttendanceTracker:
    def __init__(self, config: Dict):
        self.config = config
        self.work_rules = config.get("work_rules", {})
        self.work_start = self.work_rules.get("work_start_time", "08:30")
        self.work_end = self.work_rules.get("work_end_time", "18:00")
        self.late_threshold = self.work_rules.get("late_threshold_minutes", 30)

    def parse_time(self, time_str: str) -> datetime:
        return datetime.strptime(time_str, "%H:%M")

    def track_attendance(self, staff_list: List[Staff],
                        check_records: List[CheckRecord],
                        date_range: tuple) -> List[AttendanceSummary]:
        summaries = []
        start_date, end_date = date_range

        staff_records = defaultdict(list)
        for record in check_records:
            staff_records[record.staff_id].append(record)

        for staff in staff_list:
            summary = self.calculate_summary(
                staff, staff_records.get(staff.staff_id, []),
                start_date, end_date
            )
            summaries.append(summary)

        return summaries

    def calculate_summary(self, staff: Staff, records: List[CheckRecord],
                          start_date: str, end_date: str) -> AttendanceSummary:
        summary = AttendanceSummary(
            staff_id=staff.staff_id,
            staff_name=staff.name,
            team=staff.team,
            date=f"{start_date} ~ {end_date}"
        )

        daily_records = defaultdict(lambda: {"check_in": None, "check_out": None})

        for record in records:
            date_key = record.check_time.strftime("%Y-%m-%d")
            if record.check_type == "check_in":
                daily_records[date_key]["check_in"] = record
            else:
                daily_records[date_key]["check_out"] = record

        current = datetime.strptime(start_date, "%Y-%m-%d")
        end = datetime.strptime(end_date, "%Y-%m-%d")

        while current <= end:
            date_key = current.strftime("%Y-%m-%d")
            summary.total_days += 1

            if date_key in daily_records:
                check_in = daily_records[date_key]["check_in"]
                check_out = daily_records[date_key]["check_out"]

                if check_in and check_in.is_valid:
                    summary.valid_check_in += 1

                    work_start = self.parse_time(self.work_start)
                    check_time = check_in.check_time.time()
                    late_minutes = (datetime.combine(check_in.check_time.date(), check_time) -
                                   datetime.combine(check_in.check_time.date(), work_start)).seconds / 60

                    if late_minutes > self.late_threshold:
                        summary.late_count += 1

                if check_out and check_out.is_valid:
                    summary.valid_check_out += 1

                    work_end = self.parse_time(self.work_end)
                    check_time = check_out.check_time.time()
                    early_minutes = (datetime.combine(check_out.check_time.date(), work_end) -
                                    datetime.combine(check_out.check_time.date(), check_time)).seconds / 60

                    if early_minutes > self.late_threshold:
                        summary.early_leave_count += 1

                if check_in and check_in.is_valid:
                    summary.actual_days += 1
            else:
                summary.absent_count += 1

            current += timedelta(days=1)

        summary.calculate_rates()
        return summary
```

### 第六步：实现巡检统计模块

```python
from typing import Dict, List
from collections import defaultdict
from models.inspection import InspectionTask

class InspectionStats:
    def __init__(self, config: Dict):
        self.config = config
        self.threshold = config.get("inspection_rules", {}).get("completion_threshold", 0.95)

    def calculate_stats(self, tasks: List[InspectionTask],
                       date_range: tuple) -> Dict:
        start_date, end_date = date_range

        filtered_tasks = [
            t for t in tasks
            if start_date <= t.planned_date <= end_date
        ]

        staff_tasks = defaultdict(list)
        for task in filtered_tasks:
            staff_tasks[task.staff_id].append(task)

        team_tasks = defaultdict(lambda: {"total": 0, "completed": 0, "cancelled": 0})

        for task in filtered_tasks:
            team_tasks[task.staff_id]["total"] += 1
            if task.status == "completed":
                team_tasks[task.staff_id]["completed"] += 1
            elif task.status == "cancelled":
                team_tasks[task.staff_id]["cancelled"] += 1

        stats = {
            "period": f"{start_date} ~ {end_date}",
            "total_tasks": len(filtered_tasks),
            "completed_tasks": sum(1 for t in filtered_tasks if t.status == "completed"),
            "pending_tasks": sum(1 for t in filtered_tasks if t.status == "pending"),
            "cancelled_tasks": sum(1 for t in filtered_tasks if t.status == "cancelled"),
            "overall_completion_rate": 0.0,
            "task_type_stats": self.calculate_type_stats(filtered_tasks),
            "staff_stats": self.calculate_staff_stats(staff_tasks),
            "below_threshold_staff": []
        }

        if stats["total_tasks"] > 0:
            stats["overall_completion_rate"] = round(
                stats["completed_tasks"] / stats["total_tasks"] * 100, 1
            )

        for staff_id, staff_stat in stats["staff_stats"].items():
            if staff_stat["completion_rate"] < self.threshold * 100:
                stats["below_threshold_staff"].append({
                    "staff_id": staff_id,
                    "completion_rate": staff_stat["completion_rate"]
                })

        return stats

    def calculate_type_stats(self, tasks: List[InspectionTask]) -> Dict:
        type_stats = defaultdict(lambda: {"total": 0, "completed": 0})

        for task in tasks:
            type_stats[task.task_type]["total"] += 1
            if task.status == "completed":
                type_stats[task.task_type]["completed"] += 1

        result = {}
        for task_type, stat in type_stats.items():
            rate = round(stat["completed"] / stat["total"] * 100, 1) if stat["total"] > 0 else 0
            result[task_type] = {
                "total": stat["total"],
                "completed": stat["completed"],
                "completion_rate": rate
            }

        return result

    def calculate_staff_stats(self, staff_tasks: Dict) -> Dict:
        result = {}

        for staff_id, tasks in staff_tasks.items():
            total = len(tasks)
            completed = sum(1 for t in tasks if t.status == "completed")

            result[staff_id] = {
                "total_tasks": total,
                "completed_tasks": completed,
                "completion_rate": round(completed / total * 100, 1) if total > 0 else 0
            }

        return result
```

from typing import Dict, List
from openpyxl import Workbook
from openpyxl.styles import Font, PatternFill, Alignment, Border, Side
from openpyxl.utils import get_column_letter
from models.check_record import AttendanceSummary

class ReportBuilder:
    def __init__(self, config: Dict):
        self.config = config

    def build_daily_report(self, attendance: List[AttendanceSummary],

        self.create_attendance_sheet(wb, attendance)
        self.create_inspection_sheet(wb, inspection_stats)

    with open("data/staff_list.json", "r", encoding="utf-8") as f:
        data = json.load(f)
    return [Staff(**item) for item in data]

def load_check_records():
    with open("data/check_records.json", "r", encoding="utf-8") as f:
        data = json.load(f)
    records = []
    for item in data:
        record = CheckRecord(
            record_id=item["record_id"],
            staff_id=item["staff_id"],
            check_type=item["check_type"],
            check_time=datetime.fromisoformat(item["check_time"]),
            location_lng=item["location_lng"],
            location_lat=item["location_lat"],
            location_name=item["location_name"],
            is_valid=item.get("is_valid", True),
            invalid_reason=item.get("invalid_reason", "")
        )
        records.append(record)
    return records

def load_inspection_tasks():
    with open("data/inspection_tasks.json", "r", encoding="utf-8") as f:
        data = json.load(f)
    tasks = []
    for item in data:
        task = InspectionTask(
            task_id=item["task_id"],
            staff_id=item["staff_id"],
            task_type=item["task_type"],
            base_station_id=item["base_station_id"],
            base_station_name=item["base_station_name"],
            planned_date=item["planned_date"],
            actual_date=item.get("actual_date"),
            status=item.get("status", "pending"),
            items=item.get("items", []),
            completed_items=item.get("completed_items", []),
            remarks=item.get("remarks", "")
        )
        tasks.append(task)
    return tasks

def main():
    print("=" * 60)
    print("代维人员考勤与巡检统计技能")
    print("=" * 60)

    config = load_config()
    staff_list = load_staff_list()
    check_records = load_check_records()
    inspection_tasks = load_inspection_tasks()

    print(f"\n已加载 {len(staff_list)} 名代维人员，")
    print(f"         {len(check_records)} 条签到记录，")
    print(f"         {len(inspection_tasks)} 条巡检任务\n")

    today = datetime.now()
    start_of_week = (today - timedelta(days=today.weekday())).strftime("%Y-%m-%d")
    end_of_week = (today + timedelta(days=6-today.weekday())).strftime("%Y-%m-%d")

    date_range = (start_of_week, end_of_week)
    print(f"统计周期：{start_of_week} ~ {end_of_week}\n")

    tracker = AttendanceTracker(config)
    attendance_summaries = tracker.track_attendance(staff_list, check_records, date_range)

    stats calculator = InspectionStats(config)
    inspection_stats = stats_calculator.calculate_stats(inspection_tasks, date_range)

    report_builder = ReportBuilder(config)
    report_builder.print_summary(attendance_summaries, inspection_stats)

    output_path = "output/代维考勤巡检统计报表.xlsx"
    Path("output").mkdir(exist_ok=True)
    report_builder.build_daily_report(attendance_summaries, inspection_stats, output_path)

    print(f"\n【报表已生成】{output_path}")

if __name__ == "__main__":
    main()
