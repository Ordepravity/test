# 实验指导书：基站告警智能分析技能

## 实验概述

本实验旨在指导学员使用 OpenClaw 平台开发一个基站告警智能分析技能，通过该技能实现对移动基站告警数据的自动分类、优先级排序和根因分析，帮助运维人员快速定位和处理网络故障。

## 实验目标

1. 掌握 OpenClaw 技能开发的基本流程
2. 学会对接第三方告警数据接口
3. 实现告警数据的智能分类算法
4. 完成告警优先级评估模型
5. 输出结构化的告警分析报告

## 实验环境

| 环境项 | 要求 |

| OpenClaw 平台 | v2.3 以上版本 |

```python
from dataclasses import dataclass
from datetime import datetime
from typing import Optional

@dataclass
class Alarm:
    alarm_id: str
    ne_type: str           # 网元类型：BSC/BTS/传输
    ne_name: str           # 网元名称
    alarm_title: str       # 告警标题
    alarm_content: str     # 告警内容
    severity: str          # 告警级别：critical/major/minor
    occur_time: datetime   # 发生时间
    clear_time: Optional[datetime] = None
    region: str = ""       # 区域
    priority_score: int = 0

    def to_dict(self):
        return {
            "alarm_id": self.alarm_id,
            "ne_type": self.ne_type,
            "ne_name": self.ne_name,
            "alarm_title": self.alarm_title,
            "alarm_content": self.alarm_content,
            "severity": self.severity,
            "occur_time": self.occur_time.isoformat(),
            "clear_time": self.clear_time.isoformat() if self.clear_time else None,
            "region": self.region,
            "priority_score": self.priority_score
        }
```

### 第三步：实现告警分类器

创建 `models/classifier.py`，实现告警智能分类：

```python
import re
from typing import List, Dict
from models.alarm import Alarm

class AlarmClassifier:
    def __init__(self, config: Dict):
        self.rules = config.get("classification_rules", {})

    def classify(self, alarm: Alarm) -> str:
        alarm_text = alarm.alarm_title + " " + alarm.alarm_content

        # 匹配关键告警
        for keyword in self.rules.get("critical", []):
            if keyword in alarm_text:
                return "critical"

        # 匹配重要告警
        for keyword in self.rules.get("major", []):
            if keyword in alarm_text:
                return "major"

        # 匹配一般告警
        for keyword in self.rules.get("minor", []):
            if keyword in alarm_text:
                return "minor"

        return "minor"  # 默认级别

    def extract_location(self, alarm: Alarm) -> str:
        pattern = r"([\u4e00-\u9fa5]+(?:市|区|县|镇))"
        match = re.search(pattern, alarm.alarm_content)
        return match.group(1) if match else "未知区域"

    def extract_equipment(self, alarm: Alarm) -> str:
        patterns = [
            r"基站名称[：:]([^\s,，]+)",
            r"网元[：:]([^\s,，]+)",
            r"小区[：:]([^\s,，]+)"
        ]
        for pattern in patterns:
            match = re.search(pattern, alarm.alarm_content)
            if match:
                return match.group(1)
        return alarm.ne_name
```

### 第四步：实现优先级评分

创建 `handlers/alarm_processor.py`，实现告警处理逻辑：

```python
from datetime import datetime
from typing import List, Dict
from models.alarm import Alarm
from models.classifier import AlarmClassifier

class AlarmProcessor:
    SEVERITY_SCORE = {
        "critical": 100,
        "major": 60,
        "minor": 20
    }

    def __init__(self, config: Dict):
        self.classifier = AlarmClassifier(config)
        self.source_weights = {
            src["name"]: src["priority_weight"]
            for src in config.get("alarm_sources", [])
        }

    def process(self, alarm: Alarm) -> Alarm:
        # 自动分类
        alarm.severity = self.classifier.classify(alarm)

        # 提取位置信息
        alarm.region = self.classifier.extract_location(alarm)

        # 计算优先级评分
        alarm.priority_score = self.calculate_priority(alarm)

        return alarm

    def calculate_priority(self, alarm: Alarm) -> int:
        base_score = self.SEVERITY_SCORE.get(alarm.severity, 20)

        # 网元类型权重
        source_weight = self.source_weights.get(alarm.ne_type, 1.0)

        # 持续时间权重（超过30分钟未恢复则加分）
        duration_weight = 1.0
        if alarm.clear_time is None:
            duration = datetime.now() - alarm.occur_time
            if duration.total_seconds() > 1800:  # 30分钟
                duration_weight = 1.5

        priority = int(base_score * source_weight * duration_weight)
        return min(priority, 100)  # 最高100分
```

### 第五步：生成分析报告

创建 `handlers/report_generator.py`：

```python
from typing import List, Dict
from collections import defaultdict
from models.alarm import Alarm

class ReportGenerator:
    def generate_summary(self, alarms: List[Alarm]) -> Dict:
        summary = {
            "total_count": len(alarms),
            "by_severity": defaultdict(int),
            "by_ne_type": defaultdict(int),
            "by_region": defaultdict(int),
            "critical_alarms": [],
            "uncleared_alarms": []
        }

        for alarm in alarms:
            summary["by_severity"][alarm.severity] += 1
            summary["by_ne_type"][alarm.ne_type] += 1
            summary["by_region"][alarm.region] += 1

            if alarm.severity == "critical":
                summary["critical_alarms"].append(alarm.to_dict())

            if alarm.clear_time is None:
                summary["uncleared_alarms"].append(alarm.to_dict())

        return summary

    def generate_recommendations(self, summary: Dict) -> List[str]:
        recommendations = []

        if summary["critical_alarms"]:
            count = len(summary["critical_alarms"])
            recommendations.append(
                f"存在 {count} 条重大告警需要立即处理，建议优先查看告警列表"
            )

        if summary["uncleared_alarms"]:
            count = len(summary["uncleared_alarms"])
            recommendations.append(
                f"有 {count} 条告警超过30分钟未恢复，请确认是否需要上门处理"
            )

        # 根据区域统计推荐
        region_stats = summary["by_region"]
        if region_stats:
            top_region = max(region_stats.items(), key=lambda x: x[1])
            recommendations.append(
                f"建议重点关注 {top_region[0]} 区域，该区域告警数量最多"
            )

        return recommendations
```

### 第六步：编写主入口

修改 `main.py`：

```python
import json
import yaml
from pathlib import Path
from models.alarm import Alarm
from handlers.alarm_processor import AlarmProcessor
from handlers.report_generator import ReportGenerator
from datetime import datetime

def load_config():
    with open("config.yaml", "r", encoding="utf-8") as f:
        return yaml.safe_load(f)

def load_sample_data():
    with open("data/sample_alarms.json", "r", encoding="utf-8") as f:
        data = json.load(f)
    alarms = []
    for item in data:
        alarm = Alarm(
            alarm_id=item["alarm_id"],
            ne_type=item["ne_type"],
            ne_name=item["ne_name"],
            alarm_title=item["alarm_title"],
            alarm_content=item["alarm_content"],
            severity="unknown",
            occur_time=datetime.fromisoformat(item["occur_time"]),
            clear_time=datetime.fromisoformat(item["clear_time"]) if item.get("clear_time") else None,
            region=""
        )
        alarms.append(alarm)
    return alarms

def main():
    print("=" * 60)
    print("基站告警智能分析技能")
    print("=" * 60)

    # 加载配置和数据
    config = load_config()
    alarms = load_sample_data()
    print(f"\n已加载 {len(alarms)} 条告警记录\n")

    # 处理告警
    processor = AlarmProcessor(config)
    report_gen = ReportGenerator()

    processed_alarms = []
    for alarm in alarms:
        processed = processor.process(alarm)
        processed_alarms.append(processed)

    # 生成报告
    summary = report_gen.generate_summary(processed_alarms)
    recommendations = report_gen.generate_recommendations(summary)

    # 输出结果
    print("【告警统计】")
    print(f"  总告警数：{summary['total_count']}")
    print(f"  重大告警：{summary['by_severity']['critical']}")
    print(f"  重要告警：{summary['by_severity']['major']}")
    print(f"  一般告警：{summary['by_severity']['minor']}")

    print("\n【未恢复告警】")
    print(f"  数量：{len(summary['uncleared_alarms'])}")

    print("\n【处理建议】")
    for i, rec in enumerate(recommendations, 1):
        print(f"  {i}. {rec}")

    print("\n【TOP5 优先处理告警】")
    sorted_alarms = sorted(processed_alarms, key=lambda x: x.priority_score, reverse=True)
    for i, alarm in enumerate(sorted_alarms[:5], 1):
        print(f"  {i}. [{alarm.severity}] {alarm.alarm_title} (评分:{alarm.priority_score})")

if __name__ == "__main__":
    main()
```

## 常见问题

| 问题 | 解决方案 |
|------|----------|
| 告警分类不准确 | 检查 `config.yaml` 中的关键词配置，确保覆盖常见告警类型 |
| 区域提取失败 | 使用正则表达式验证告警内容格式 |
| 评分异常 | 确认 `priority_weight` 参数在合理范围内（0.5-2.0） |
